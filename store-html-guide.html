🐬 Pure HTML बाट Store use गर्ने तरिका

कुनै JavaScript नचाहिने — सबै data-* attributes बाट!

🧠 कसरी काम गर्छ?

Dolphin मा २ प्रकारका stores छन्:

Store के हो? HTML tag
UI Store (app) Local state — search query, selected filter, modal open/close <dolphin-store name="app">
DB Store (products) Server/API data — real-time, filter, sort, search data-api-get="..." data-api-store="app.products"

✅ Complete Example — Search + Filter + Sort + Delete

<!-- ① UI Store सुरु गर्ने — search query, category, sort state राख्न -->
<dolphin-store
  name="app"
  searchQuery=""
  selectedCategory=""
  sortBy="">
</dolphin-store>

<!-- ② Search Input -->
<input
  type="text"
  placeholder="Search products..."
  data-store-write="app.searchQuery"
  data-store-input="products.search(app.searchQuery, ['title', 'description'])"
/>

<!-- ③ Category Filter -->
<select
  data-store-write="app.selectedCategory"
  data-store-change="products.filter('category', app.selectedCategory)">
  <option value="">All Categories</option>
  <option value="electronics">Electronics</option>
  <option value="jewelery">Jewelery</option>
  <option value="men's clothing">Men's Clothing</option>
</select>

<!-- ④ Sort Buttons -->
<button data-store-click="products.sort('price', true)">Price ↑</button>
<button data-store-click="products.sort('price', false)">Price ↓</button>
<button data-store-click="products.sort('title', true)">Name A→Z</button>
<button data-store-click="products.clearFilters()">Clear All</button>

<!-- ⑤ Product Grid — DB Store बाट data fetch हुन्छ -->
<div
  data-api-get="https://fakestoreapi.com/products"
  data-api-store="app.products"
  data-rt-bind="store/products"
  data-rt-template="#product-card">
  <!-- Loading spinner -->
  <div>Loading...</div>
</div>

<!-- ⑥ Product Card Template -->
<template id="product-card">
  {#each items as item}
    <div class="card">
      <img src="{{item.image}}" alt="{{item.title}}" />
      <h3>{{item.title}}</h3>
      <span>$\{{item.price}}</span>

      <!-- Delete Button — optimistic (तुरुन्त हटाउने) -->
      <button data-store-click="
        products.optimisticDelete(
          item.id,
          () => fetch('/api/products/' + item.id, { method: 'DELETE' })
        )
      ">🗑 Delete</button>
    </div>
  {/each}
</template>

📚 Feature-by-Feature Guide

① Search (खोज्ने)

<!-- data-store-input: typing हुँदा automatically search हुन्छ -->
<input
  data-store-write="app.searchQuery"
  data-store-input="products.search(app.searchQuery, ['title', 'description'])"
  placeholder="Search..."
/>

<!-- fields नदिए सबै fields मा search हुन्छ -->
<input
  data-store-input="products.search(app.searchQuery)"
  placeholder="Search all fields..."
/>

② Filter (छान्ने)

<!-- Select dropdown बाट filter -->
<select
  data-store-write="app.selectedCategory"
  data-store-change="products.filter('category', app.selectedCategory)">
  <option value="">All</option>
  <option value="electronics">Electronics</option>
</select>

<!-- Button बाट fixed value filter -->
<button data-store-click="products.filter('inStock', true)">In Stock Only</button>

<!-- Filter हटाउन -->
<button data-store-click="products.filter('category', null)">Show All</button>

③ Range Filter (मूल्य दायरा)

<dolphin-store name="app" minPrice="0" maxPrice="1000"></dolphin-store>

<input type="range" min="0" max="500"
  data-store-write="app.maxPrice"
  data-store-input="products.range('price', app.minPrice, app.maxPrice)"
/>

<!-- Fixed buttons -->
<button data-store-click="products.range('price', 0, 50)">Under $50</button>
<button data-store-click="products.range('price', 50, 200)">$50–$200</button>

④ Sort (मिलाउने)

<!-- Dropdown बाट sort -->
<select data-store-write="app.sortBy"
        data-store-change="
          app.sortBy === 'price-low'  ? products.sort('price', true)  :
          app.sortBy === 'price-high' ? products.sort('price', false) :
          app.sortBy === 'title'      ? products.sort('title', true)  :
          products.clearFilters()
        ">
  <option value="">Default</option>
  <option value="price-low">Price: Low to High</option>
  <option value="price-high">Price: High to Low</option>
  <option value="title">Name A→Z</option>
</select>

<!-- Buttons -->
<button data-store-click="products.sort('price', true)">Price ↑</button>
<button data-store-click="products.sort('price', false)">Price ↓</button>

⑤ Clear All Filters

<button data-store-click="
  products.clearFilters();
  app.searchQuery = '';
  app.selectedCategory = '';
  app.sortBy = '';
">🗑 Reset Everything</button>

⑥ Delete (हटाउने)

<!-- Template भित्रको item को id use गरेर delete -->
<template id="product-card">
  {#each items as item}
    <div>
      {{item.title}}

      <!-- Instant delete (optimistic) — API fail भए rollback -->
      <button data-store-click="
        products.optimisticDelete(
          item.id,
          () => fetch('/api/products/' + item.id, { method: 'DELETE' })
        )
      ">Delete</button>

      <!-- Simple delete (without API) -->
      <button data-store-click="products.deleteById(item.id)">Remove</button>
    </div>
  {/each}
</template>

⑦ Multiple Operations एकैसाथ

<!-- Semicolon बाट multiple actions -->
<button data-store-click="
  products.filter('category', 'electronics');
  products.sort('price', true);
  products.range('price', 0, 500)
">Electronics under $500, by price</button>

🎨 Template मा items कसरी render हुन्छ?

<!-- data-rt-bind="store/products" = products collection बाट data render गर्ने -->
<!-- data-rt-template="#myTemplate" = यो template use गर्ने -->

<div
  data-api-get="/api/products"
  data-api-store="app.products"
  data-rt-bind="store/products"
  data-rt-template="#product-card">
</div>

<template id="product-card">
  <!-- items = filtered + sorted results हो -->
  {#each items as item}
    <div>{{item.title}} — $\{{item.price}}</div>
  {/each}

  <!-- items खाली भएको बेला -->
  {#if items.length === 0}
    <p>No results found.</p>
  {/if}
</template>
⚠️ Note: Template भित्र items = filtered/sorted results। item.id, item.title etc. directly use गर्न सकिन्छ।
