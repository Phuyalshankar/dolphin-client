🐬 Dolphin Store v2 — HTML Usage Examples

① Setup

<script src="./dolphin-client.js"></script>
<script>
  const client = new DolphinClient('http://localhost:3000');
  // client.store अब automatically available छ!
</script>

② Search (खोज्ने)

// JavaScript मा:
const searchInput = document.getElementById('searchInput');
searchInput.addEventListener('input', (e) => {
  // products collection मा search लाउने
  client.store.products.search(e.target.value, ['title', 'description']);
  renderProducts();
});

function renderProducts() {
  const products = client.store.products.items; // filtered results
  document.getElementById('productList').innerHTML =
    products.map(p => `<div>${p.title} - $${p.price}</div>`).join('');
}

③ Filter (छान्ने)

document.getElementById('categoryFilter').addEventListener('change', (e) => {
  // category field मा exact match filter
  client.store.products.filter('category', e.target.value);
  renderProducts();
});

// Filter हटाउन:
client.store.products.filter('category', null); // वा ''
renderProducts();

④ Range Filter (मूल्य अनुसार)

// $10 देखि $100 सम्म मात्र देखाउने
client.store.products.range('price', 10, 100);
renderProducts();

⑤ Sort (मिलाउने)

document.getElementById('sortPriceAsc').onclick = () => {
  client.store.products.sort('price', true);  // true = ascending
  renderProducts();
};

document.getElementById('sortPriceDesc').onclick = () => {
  client.store.products.sort('price', false); // false = descending
  renderProducts();
};

document.getElementById('sortTitle').onclick = () => {
  client.store.products.sort('title', true);
  renderProducts();
};

// सबै filters र sort हटाउन:
document.getElementById('clearAll').onclick = () => {
  client.store.products.clearFilters();
  renderProducts();
};

⑥ Pagination (पेजिनेशन)

let currentPage = 1;
const PAGE_SIZE = 5;

function renderPage() {
  // page() ले filtered data बाट paginate गर्छ
  const result = client.store.products.page(currentPage, PAGE_SIZE);
  // result = { data: [...], total: 20, page: 1, size: 5, pages: 4, hasNext: true, hasPrev: false }

  document.getElementById('paginatedList').innerHTML =
    result.data.map(p => `<div class="card">${p.title}</div>`).join('');

  document.getElementById('pageInfo').textContent =
    `Page ${result.page} of ${result.pages} (${result.total} total)`;

  document.getElementById('prevPage').disabled = !result.hasPrev;
  document.getElementById('nextPage').disabled = !result.hasNext;
}

document.getElementById('prevPage').onclick = () => { currentPage--; renderPage(); };
document.getElementById('nextPage').onclick = () => { currentPage++; renderPage(); };

⑦ Optimistic Delete (तुरुन्त हटाउने)

// Button click मा:
async function deleteProduct(id) {
  try {
    await client.store.products.optimisticDelete(
      id,                                          // हटाउने item को id
      () => client.api.request('DELETE', `/products/${id}`)  // API call
    );
    // ✅ UI तुरुन्त update हुन्छ (API finish हुन कुर्दैन)
    // ✅ API fail भए UI rollback हुन्छ
    console.log('Deleted!');
  } catch (err) {
    alert('Delete failed! UI restored.');
  }
}

// HTML मा:
// <button onclick="deleteProduct({{item.id}})">Delete</button>

⑧ Optimistic Update (तुरुन्त update)

async function updateProduct(id, newPrice) {
  try {
    await client.store.products.optimisticUpdate(
      id,                                         // update गर्ने item को id
      { price: newPrice },                        // नयाँ data
      () => client.api.request('PUT', `/products/${id}`, { price: newPrice })
    );
    // ✅ UI तुरुन्त update — नयाँ price देखिन्छ
    // ✅ API fail भए पुरानो price फर्कन्छ
  } catch (err) {
    alert('Update failed! Price restored.');
  }
}

⑨ Per-Item Loading (item-level spinner)

// Delete button मा per-item spinner देखाउन:
async function deleteWithSpinner(id) {
  // Loading सुरु गर्ने
  client.store.products.trackStart(id);
  renderProducts(); // spinner देखिन्छ

  try {
    await client.api.request('DELETE', `/products/${id}`);
    // Real delete after API success
    client.store.products.deleteById(id);
  } finally {
    // Loading बन्द गर्ने
    client.store.products.trackEnd(id);
    renderProducts();
  }
}

// Render मा spinner:
function renderProducts() {
  const items = client.store.products.items;
  document.getElementById('list').innerHTML = items.map(item => `
    <div>
      ${item.title}
      ${client.store.products.isLoading(item.id)
        ? '<span class="spinner">⏳</span>'  // ← यो item loading छ
        : `<button onclick="deleteWithSpinner(${item.id})">Delete</button>`
      }
    </div>
  `).join('');
}

⑫ Complete Working Example

<!DOCTYPE html>
<html>
<body>

  <!-- Search + Filter UI -->
  <input id="q" placeholder="Search..." oninput="handleSearch(this.value)">
  <select onchange="handleCategory(this.value)">
    <option value="">All</option>
    <option value="electronics">Electronics</option>
  </select>
  <button onclick="handleSort()">Sort by Price</button>
  <button onclick="handleClear()">Clear</button>

  <!-- Product List -->
  <div id="list"></div>

  <!-- Pagination -->
  <button onclick="prevPage()">← Prev</button>
  <span id="pageInfo"></span>
  <button onclick="nextPage()">Next →</button>

  <script src="./dolphin-client.js"></script>
  <script>
    const client = new DolphinClient('http://localhost:3000');
    let page = 1;
    let sortAsc = true;

    // Store subscribe — data बदलिए automatically render
    client.store.subscribe(() => render());

    // Data fetch हुन्छ automatically
    const _ = client.store.products; // access गर्नासाथ fetch सुरु

    function handleSearch(val) {
      client.store.products.search(val, ['title', 'description']);
      page = 1;
    }

    function handleCategory(val) {
      client.store.products.filter('category', val);
      page = 1;
    }

    function handleSort() {
      sortAsc = !sortAsc;
      client.store.products.sort('price', sortAsc);
    }

    function handleClear() {
      client.store.products.clearFilters();
      page = 1;
    }

    async function deleteProduct(id) {
      client.store.products.trackStart(id);
      try {
        await client.store.products.optimisticDelete(
          id,
          () => client.api.request('DELETE', `/products/${id}`)
        );
      } catch {
        alert('Delete failed!');
      } finally {
        client.store.products.trackEnd(id);
      }
    }

    function render() {
      const result = client.store.products.page(page, 8);

      document.getElementById('list').innerHTML = result.data.map(p => `
        <div>
          <strong>${p.title}</strong> — $${p.price}
          ${client.store.products.isLoading(p.id)
            ? '<span>⏳ Deleting...</span>'
            : `<button onclick="deleteProduct(${p.id})">🗑 Delete</button>`}
        </div>
      `).join('');

      document.getElementById('pageInfo').textContent =
        `Page ${result.page}/${result.pages} (${result.total} items)`;
    }

    function prevPage() { if (page > 1) { page--; render(); } }
    function nextPage() {
      const r = client.store.products.page(page, 8);
      if (r.hasNext) { page++; render(); }
    }
  </script>
</body>
</html>
